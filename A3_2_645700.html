ALLOWED_EXTS = (".png", ".jpg", ".jpeg", ".bmp", ".tif", ".tiff")

def infer_label_from_path(path):
    base = os.path.basename(path)
    m = re.search(r'(?<!\d)(\d)(?!\d)', base)
    if m:
        return int(m.group(1))
    parent = os.path.basename(os.path.dirname(path))
    if parent.isdigit() and len(parent)==1:
        return int(parent)
    return None

def preprocess_to_mnist(gray_img, target_size=28):
    # 1) Otsu + invert (dgito blanco sobre fondo negro)
    _, bin_inv = cv2.threshold(gray_img, 0, 255, cv2.THRESH_BINARY_INV | cv2.THRESH_OTSU)
    # Mejora 1) Recorte al contorno mayor
    cnts, _ = cv2.findContours(bin_inv, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if len(cnts)==0:
        crop = bin_inv
    else:
        c = max(cnts, key=cv2.contourArea)
        x,y,w,h = cv2.boundingRect(c)
        crop = bin_inv[y:y+h, x:x+w]
    # Mejora  2) Cuadrar con padding
    h,w = crop.shape
    side = max(h,w)
    pad_t = (side-h)//2; pad_b = side-h-pad_t
    pad_l = (side-w)//2; pad_r = side-w-pad_l
    square = cv2.copyMakeBorder(crop, pad_t, pad_b, pad_l, pad_r, borderType=cv2.BORDER_CONSTANT, value=0)
    # Mejora 4) Escalar y normalizar
    resized = cv2.resize(square, (target_size, target_size), interpolation=cv2.INTER_AREA)
    return resized.astype(np.float32)/255.0

def prepare_for_model(X_28x28, model):
    """ Ajusta forma segn el modelo activo (MLP espera (N,28,28), CNN (N,28,28,1)) """
    if len(model.input_shape)==4:
        return X_28x28[..., np.newaxis]
    return X_28x28
